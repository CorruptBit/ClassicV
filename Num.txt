  private void OnCollisionEnter(Collision collision)
    {
       

            if (PlayerPV.IsMine == false)
            {
                return;
            }




            if (collision.transform.tag == "OBS")
            {


                StopAllCoroutines();
            MStopCoroutine();
                CollideFX(collision.transform.localPosition);
            if(BoostActive)
            {
                StartCoroutine(ConsumingStamina());
            }


                COLLIDESFXFUNX();

                SpeedController = 0.5f;

                if (collision.gameObject.GetComponent<PhotonView>() != null)
                {



                    if (PhotonNetwork.IsMasterClient)
                {
                    PhotonNetwork.Destroy(collision.gameObject);
                    MyHealth--;
                    NM.GotHit(MyHealth);
                    GameObject.FindWithTag("MGLM").GetComponent<MultiplayerGameLevelManger>().LifeUpdate(MyHealth, true);
                    Movement_Coroutine = StartCoroutine("AfterCollision");
                }




                    else
                    {
                    collision.gameObject.SetActive(false);
                        NM.VD(collision.gameObject.GetComponent<PhotonView>().ViewID);
                    MyHealth--;
                    NM.GotHit(MyHealth);
                    GameObject.FindWithTag("MGLM").GetComponent<MultiplayerGameLevelManger>().LifeUpdate(MyHealth, true);
                    Movement_Coroutine = StartCoroutine("AfterCollision");
                    Vibration.Vibrate(250);
                }


                }

                else
                {
                    if (collision.transform.parent != null && collision.transform.parent.tag != "Sets" && collision.transform.parent.tag == "OBS" && collision.transform.parent.gameObject.GetComponent<PhotonView>() != null)
                    {

                    StartCoroutine(RechargeStaminaOverTime());

                        if (PhotonNetwork.IsMasterClient)
                        {
                            PhotonNetwork.Destroy(collision.transform.parent.gameObject);
                        MyHealth--;
                        NM.GotHit(MyHealth);
                        GameObject.FindWithTag("MGLM").GetComponent<MultiplayerGameLevelManger>().LifeUpdate(MyHealth, true);
                    }
                        else
                        {
                        collision.transform.parent.gameObject.SetActive(false);

                            NM.VD(collision.transform.parent.gameObject.GetComponent<PhotonView>().ViewID);
                        MyHealth--;
                        NM.GotHit(MyHealth);
                        GameObject.FindWithTag("MGLM").GetComponent<MultiplayerGameLevelManger>().LifeUpdate(MyHealth, true);
                        Vibration.Vibrate(250);
                    }
                
                
                
                
                }
                    else
                    {
                        Destroy(collision.gameObject);
                        //Simply Destroy Locally...
                    }
                }

            if (MyHealth < 1)
            {
                PAnim.SetBool("Collided", true);
                GetComponent<IM>().enabled = false;
                SpeedController = 0;
               
                
                NM.GameFinished(NM.OtherPlayerID); //Its an RPC call for everyBody Winner is based on Players Ids...
                //Call the other to end the game....
               // GameObject.FindWithTag("MGLM").GetComponent<MultiplayerGameLevelManger>().GameFinished(false);

               
            }
        }



        




    }
